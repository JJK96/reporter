%Issue environment
\newenvironment{issue}[3][]{
    \stepcounter{issue}
    \newcommand{\@issuetitle}{#2}
    \newcommand{\@issuenumber}{#3}
    \def\labeltmp{#1}
    \ifx\labeltmp\@empty%
    \else%
        \mylabeltext{\VAR{text.finding} \@issuenumber{}}{\labeltmp}
    \fi
    \mylabeltext{\VAR{text.finding} \@issuenumber{}}{finding-\@issuenumber}
    \subsubsection{\@issuetitle}
}{
    ~\par
    {\tablestyle[default]
    \disablealternatecolors
    \begin{tabularx}{\textwidth}{|>{\bfseries}l|X|}
        \theadstart
        \multicolumn{2}{|l|}{\textbf{\VAR{text.finding} \@issuenumber{}:}~\@issuetitle} \\
        \tbody
        \issueheader
        \hline
        \VAR{text.description} & \@description \tnl
        \VAR{text.solution} & \@solution \tnl
        \VAR{text.location} & \@location \tnl
        \VAR{text.cvss_vector} & \@cvssvector \tnl
        \tend
    \end{tabularx}
    \par}
}
